  completed_at DATE NOT NULL DEFAULT CURRENT_DATE,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),

  to_node_id UUID REFERENCES nodes(id) ON DELETE CASCADE,
  type TEXT NOT NULL DEFAULT 'enhances',

  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
